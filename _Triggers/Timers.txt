Trigger("\x006Kings"){
Conditions:
	Switch("Start", set);
	Deaths("Current Player", "Pay Resources Switch", at most, 0);

Actions:
	Set Deaths("Current Player", "Bengalaas (Jungle)", Subtract, 1);
	Preserve Trigger();
	Comment("Timer, 12 per second");
}
